    Status changer:
    ```typescript
    const initiateStatusChange = (newStatus: MeetingStatus) => {
      setPendingStatus(newStatus)
      setStatusConfirmOpen(true)
    }
    const confirmStatusChange = () => {
      if (!pendingStatus || !meeting) return
      setChangingStatus(true)
      const prev = meeting.status
      setMeeting({ ...meeting, status: pendingStatus })  // optimistic
      fetch(`/api/meetings/${id}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ status: pendingStatus, prev_status: prev }),
      }).then(() => {
        setChangingStatus(false)
        setStatusConfirmOpen(false)
      })
      // fire-and-forget pattern (optimistic)
    }
    ```
