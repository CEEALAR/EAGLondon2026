  email: string
  display_name: string | null
  avatar_url: string | null

<!-- New type exported from meetings-timeline.tsx (used by page.tsx) -->
export type TimelineMeeting = {
  id: string
  status: MeetingStatus
  scheduled_at: string        // ISO string, e.g. '2026-05-29T09:00:00+00:00'
  location: string | null
  owner_id: string | null
  attendee_name: string       // joined from attendees(first_name, last_name)
}

<!-- Supabase query in page.tsx — join attendees table for name -->
supabase
  .from('meetings')
  .select('id, status, scheduled_at, location, owner_id, attendees(first_name, last_name)')
  .not('scheduled_at', 'is', null)
  .gte('scheduled_at', '2026-05-29T00:00:00+00:00')
  .lte('scheduled_at', '2026-05-31T23:59:59+00:00')
  .order('scheduled_at')

  <action>
    Create ceealar-pulse/app/(app)/meetings/_components/meetings-timeline.tsx as a Client Component
    ('use client' directive at the top). The component exports both MeetingsTimeline and the
    TimelineMeeting type.

    Constants:
    - CONF_DAYS: array of { value: string, label: string } for '2026-05-29' (29 May),
      '2026-05-30' (30 May), '2026-05-31' (31 May)
    - HOUR_START = 8 (08:00 first rendered hour)
    - HOUR_END = 22 (exclusive — renders 08:00 through 21:xx)
    - HOUR_HEIGHT = 60 (pixels per hour; 1 px = 1 minute, 30-min meetings = 29px block height)

    statusStyle(s: MeetingStatus): string — returns Tailwind class string:
    - 'planned'      → 'bg-[#0F766E] text-white'
    - 'done'         → 'bg-[#D4A017] text-white'
    - 'want_to_meet' → 'bg-gray-100 text-gray-700 border border-gray-300'
    - default        → 'bg-muted text-muted-foreground border border-border'

    MeetingBlock({ meeting }: { meeting: TimelineMeeting }):
    - Parse meeting.scheduled_at with parseISO from date-fns.
    - Compute topPx: (date.getHours() - HOUR_START) * HOUR_HEIGHT + date.getMinutes()
    - Render a next/link wrapping an absolutely-positioned div:
        className: combines statusStyle + 'absolute left-0.5 right-0.5 rounded px-1 py-0.5
          text-[11px] leading-tight overflow-hidden hover:opacity-80 transition-opacity'
        style: { top: topPx + 'px', height: '29px' }
    - Inside: <p className="font-medium truncate">{meeting.attendee_name}</p>
    - If meeting.location: <p className="truncate opacity-75 text-[10px]">{meeting.location}</p>

    MemberLane({ meetings }: { meetings: TimelineMeeting[] }):
    - Outer div: relative, height = (HOUR_END - HOUR_START) * HOUR_HEIGHT px
    - Render hour grid lines: Array.from({ length: HOUR_END - HOUR_START }) — for each index i:
        absolute div with border-t border-border at top: i * HOUR_HEIGHT px
        Inside: span with time label (HOUR_START + i, zero-padded to 2 digits)
        Label span classes: 'absolute -top-2.5 left-0 text-[10px] text-muted-foreground
          select-none w-7 text-right pr-0.5'
    - Meeting blocks container: absolute inset-0 with left: 28px (clears time label width)
    - Render a MeetingBlock for each meeting

    MeetingsTimeline({ meetings, teamMembers }):
    State:
    - selectedDay: string — initialized to CONF_DAYS[0].value
    - selectedMemberId: string — initialized to teamMembers[0]?.id ?? ''

    Data grouping (useMemo, deps [meetings, teamMembers]):
    - Build map: Record<string, Record<string, TimelineMeeting[]>>
      Outer key: day value. Inner key: team member id. Initialize all combinations to [].
    - For each meeting m: day = m.scheduled_at.slice(0, 10)
      If map[day] exists and m.owner_id exists and map[day][m.owner_id] exists:
        push m into that array.

    JSX structure:
    1. Day tab bar: flex gap-0 border-b mb-4
       Each button: px-4 py-2 text-sm font-medium border-b-2 -mb-px transition-colors
       Active: 'border-primary text-primary' | Inactive: 'border-transparent text-muted-foreground hover:text-foreground'

    2. Mobile section (md:hidden):
       - Member pill tab row: flex gap-1.5 overflow-x-auto pb-2 mb-3
         Each button: px-3 py-1 text-xs font-medium rounded-full whitespace-nowrap flex-shrink-0 transition-colors
         Active: 'bg-primary text-primary-foreground' | Inactive: 'bg-muted text-muted-foreground'
         Label: m.display_name ?? m.email.split('@')[0]
       - overflow-y-auto div containing MemberLane for selectedMemberId

    3. Desktop section (hidden md:grid md:grid-cols-4 gap-2):
       For each teamMember:
         - Member name heading: text-xs font-semibold text-center mb-2 text-foreground truncate
         - MemberLane for that member's meetings in the selected day
